   156→def _preflight_screen_slot(slot_index: int) -> Tuple[bool, str]:
   157→    """
   158→    Enhanced pre-flight screening for Cloudflare slots (Amendment 6).
   159→
   160→    Validates account_id format (32-char hex), API token length (>=40 chars),
   161→    and gateway URL structure BEFORE sending any request.
   162→
   163→    Broken slots (like slot 7 with corrupted tokens) are detected here
   164→    and silently skipped without causing HTTP 400 errors.
   165→
   166→    Returns:
   167→        (valid: bool, reason: str) — valid=True means slot is usable.
   168→    """
   169→    account_id  = os.environ.get(f"CF_ACCOUNT_ID_{slot_index}", "").strip()
   170→    api_token   = os.environ.get(f"CF_API_TOKEN_{slot_index}", "").strip()
   171→    gateway_url = os.environ.get(f"CF_AI_GATEWAY_URL_{slot_index}", "").strip()
   172→
   173→    # Rule 1: Both must be non-empty
   174→    if not account_id or not api_token:
   175→        return False, "missing credentials"
   176→
   177→    # Rule 2: Account ID must be 32-char hex
   178→    if not re.match(r'^[0-9a-f]{32}$', account_id, re.IGNORECASE):
   179→        return False, f"invalid account_id format (len={len(account_id)})"
   180→
   181→    # Rule 3: API token must be >=40 chars (CF tokens are 40+)
   182→    if len(api_token) < CF_MIN_TOKEN_LENGTH:
   183→        return False, f"token too short ({len(api_token)} chars, min={CF_MIN_TOKEN_LENGTH})"
   184→
   185→    # Rule 4: Gateway URL must match CF pattern (if provided)
   186→    if gateway_url:
   187→        pattern = r'^https://gateway\.ai\.cloudflare\.com/v1/[0-9a-f]{32}/'
   188→        if not re.match(pattern, gateway_url, re.IGNORECASE):
   189→            return False, "malformed gateway URL"
   190→
   191→    # Rule 5: Account ID in gateway URL must match credentials
   192→    if gateway_url and account_id:
   193→        # Extract account_id from gateway URL
   194→        gw_match = re.search(r'/v1/([0-9a-f]{32})/', gateway_url, re.IGNORECASE)
   195→        if gw_match:
   196→            gw_acct = gw_match.group(1)
   197→            if gw_acct.lower() != account_id.lower():
   198→                return False, "account_id mismatch between URL and credentials"
   199→
   200→    return True, "ok"

   203→def preflight_validate_cf_slot(
   204→    slot_index: int,
   205→    account_id: str,
   206→    api_token: str,
   207→    gateway_url: str = "",
   208→) -> list:
   209→    """
   210→    Pre-flight screening for Cloudflare slots — validates token length,
   211→    account_id format, and gateway URL structure BEFORE sending any request.
   212→
   213→    Broken slots (like slot 7 with corrupted tokens) are detected here
   214→    and silently skipped without causing HTTP 400 errors.
   215→
   216→    Returns a list of warning strings (empty = slot looks valid).
   217→    The slot is NOT removed — only flagged for skipping at runtime.
   218→    """
   219→    issues = []
   220→
   221→    # Validate API token
   222→    if not api_token:
   223→        issues.append(f"Slot {slot_index}: CF_API_TOKEN is empty")
   224→    else:
   225→        clean_token = api_token.strip()
   226→        if len(clean_token) < CF_MIN_TOKEN_LENGTH:
   227→            issues.append(
   228→                f"Slot {slot_index}: CF_API_TOKEN too short "
   229→                f"(len={len(clean_token)}, min={CF_MIN_TOKEN_LENGTH}). "
   230→                f"Token appears corrupted or incomplete."
   231→            )
   232→        if len(clean_token) > CF_MAX_TOKEN_LENGTH:
   233→            issues.append(
   234→                f"Slot {slot_index}: CF_API_TOKEN too long "
   235→                f"(len={len(clean_token)}, max={CF_MAX_TOKEN_LENGTH}). "
   236→                f"Token may contain extra characters or multiple tokens."
   237→            )
   238→        if '\n' in clean_token or '\r' in clean_token:
   239→            issues.append(
   240→                f"Slot {slot_index}: CF_API_TOKEN contains newline characters — "
   241→                f"possible copy-paste error from GitHub Secrets"
   242→            )
   243→        # CF API tokens should not have spaces
   244→        if ' ' in clean_token:
   245→            issues.append(
   246→                f"Slot {slot_index}: CF_API_TOKEN contains spaces — "
   247→                f"token is likely corrupted"
   248→            )
   249→
   250→    # Validate account ID — must be 32-char hex
   251→    if not account_id:
   252→        issues.append(f"Slot {slot_index}: CF_ACCOUNT_ID is empty")
   253→    else:
   254→        clean_acct = account_id.strip()
   255→        if not re.match(r'^[0-9a-f]{32}$', clean_acct, re.IGNORECASE):
   256→            issues.append(
   257→                f"Slot {slot_index}: CF_ACCOUNT_ID invalid format "
   258→                f"(len={len(clean_acct)}, expected 32-char hex). "
   259→                f"Account ID appears corrupted."
   260→            )
   261→
   262→    # Validate gateway URL (only for AI Gateway provider)
   263→    if gateway_url:
   264→        if not gateway_url.startswith("https://"):
   265→            issues.append(
   266→                f"Slot {slot_index}: CF_AI_GATEWAY_URL does not start with https://"
   267→            )
   268→        else:
   269→            # Must match CF AI Gateway pattern
   270→            pattern = r'^https://gateway\.ai\.cloudflare\.com/v1/[0-9a-f]{32}/'
   271→            if not re.match(pattern, gateway_url, re.IGNORECASE):
   272→                issues.append(
   273→                    f"Slot {slot_index}: CF_AI_GATEWAY_URL malformed — "
   274→                    f"must match https://gateway.ai.cloudflare.com/v1/{{account_id}}/{{slug}}"
   275→                )
   276→            else:
   277→                # Check account_id in URL matches credentials
   278→                gw_match = re.search(r'/v1/([0-9a-f]{32})/', gateway_url, re.IGNORECASE)
   279→                if gw_match and account_id:
   280→                    gw_acct = gw_match.group(1)
   281→                    if gw_acct.lower() != account_id.strip().lower():
   282→                        issues.append(
   283→                            f"Slot {slot_index}: Account ID in gateway URL "
   284→                            f"({_mask_key(gw_acct, 3)}...) does not match "
   285→                            f"CF_ACCOUNT_ID ({_mask_key(account_id, 3)}...)"
   286→                        )
   287→
   288→    if issues:
   289→        for issue in issues:
   290→            logger.warning(f"[CF-Preflight] {issue}")
   291→        logger.warning(
   292→            f"[CF-Preflight] Slot {slot_index} FAILED pre-flight screening — "
   293→            f"will be silently skipped (NOT deleted from config). "
   294→            f"{len(issues)} issue(s) detected."
   295→        )
   296→    else:
   297→        logger.debug(
   298→            f"[CF-Preflight] Slot {slot_index} PASSED pre-flight screening "
   299→            f"(token_len={len(api_token.strip())}, "
   300→            f"acct_id_len={len(account_id.strip())})"
   301→        )
   302→
   303→    return issues

   350→def _mask_key(key: str, visible: int = 4) -> str:
   351→    """Mask sensitive key for logging, showing only first/last chars."""
   352→    if not key:
   353→        return "<EMPTY>"
   354→    if len(key) <= visible * 2:
   355→        return f"{key[:2]}***{key[-2:]}" if len(key) >= 4 else "***"
   356→    return f"{key[:visible]}...{key[-visible:]}"

   359→def _validate_url(url: str, label: str) -> str:
   360→    if not url.startswith("https://"):
   361→        raise ValueError(
   362→            f"[{label}] Invalid URL '{url[:40]}': must be absolute HTTPS."
   363→        )
   364→    return url.rstrip("/")

   367→def _compute_backoff_delay(attempt: int) -> float:
   368→    """Compute exponential backoff delay with jitter."""
   369→    raw = RETRY_BASE_DELAY_SEC * (2 ** attempt)
   370→    jittered = raw + random.uniform(-RETRY_JITTER_SEC, RETRY_JITTER_SEC)
   371→    return min(max(jittered, 0.1), RETRY_MAX_DELAY_SEC)

   374→def _sanitize_api_key(key: str) -> str:
   375→    """Sanitize API key: strip whitespace, newlines, and null bytes."""
   376→    if not key:
   377→        return ""
   378→    cleaned = key.strip().replace("\n", "").replace("\r", "").replace("\0", "")
   379→    if cleaned != key:
   380→        logger.warning(
   381→            f"API key had trailing whitespace/newlines — sanitized "
   382→            f"(original length={len(key)}, cleaned={len(cleaned)})"
   383→        )
   384→    return cleaned

   395→def _log_auth_failure(
   396→    provider: str,
   397→    slot_index: int,
   398→    error: urllib.error.HTTPError,
   399→    url: str,
   400→    headers_sent: dict,
   401→):
   402→    """
   403→    Log verbose diagnostic information for 403/400 errors.
   404→    Masks all sensitive keys. Only called on auth failures.
   405→    """
   406→    error_body = _read_error_body(error)
   407→
   408→    # Mask headers
   409→    sensitive_keys = {
   410→        "authorization", "x-portkey-api-key", "api-key",
   411→        "x-api-key", "bearer", "token",
   412→    }
   413→    masked_headers = {}
   414→    for k, v in headers_sent.items():
   415→        if k.lower() in sensitive_keys:
   416→            masked_headers[k] = _mask_key(str(v))
   417→        else:
   418→            masked_headers[k] = str(v)
   419→
   420→    logger.error(
   421→        f"[{provider}] slot {slot_index} AUTH FAILURE: "
   422→        f"HTTP {error.code} {error.reason}"
   423→    )
   424→    logger.error(f"  URL: {_mask_url(url)}")
   425→    logger.error(f"  Headers: {masked_headers}")
   426→    logger.error(f"  Response body: {error_body[:300]}")
   427→
   428→    # Infer root cause
   429→    if error.code == 403:
   430→        body_lower = error_body.lower()
   431→        # Detect Cloudflare bot protection (error code 1010)
   432→        if _CF_BOT_ERROR_CODE in error_body:
   433→            logger.error(
   434→                f"  DIAGNOSIS: CLOUDFLARE_BOT_PROTECTION — request blocked by "
   435→                f"Cloudflare anti-bot (error code 1010). This is NOT an auth failure. "
   436→                f"The User-Agent header may be missing or blocked. "
   437→                f"The request will be retried with backoff."
   438→            )
   439→        elif "invalid" in body_lower or "unauthorized" in body_lower:
   440→            logger.error(f"  DIAGNOSIS: INVALID_CREDENTIALS — key rejected by provider")
   441→        elif "quota" in body_lower or "limit" in body_lower or "rate" in body_lower:
   442→            logger.error(f"  DIAGNOSIS: QUOTA_EXCEEDED — account has hit limits")
   443→        elif "sanction" in body_lower or "region" in body_lower or "embargo" in body_lower:
   444→            logger.error(f"  DIAGNOSIS: REGION_BLOCKED — provider blocks this region/IP")
   445→        elif "expired" in body_lower:
   446→            logger.error(f"  DIAGNOSIS: KEY_EXPIRED — API key has expired")
   447→        else:
   448→            logger.error(
   449→                f"  DIAGNOSIS: AUTH_FAILURE — likely invalid/expired key "
   450→                f"or insufficient permissions. Check key format, whitespace, and account status."
   451→            )
   452→    elif error.code == 400:
   453→        body_lower = error_body.lower()
   454→        if not error_body.strip():
   455→            logger.error(
   456→                f"  DIAGNOSIS: EMPTY_RESPONSE_BODY_400 — server returned 400 with empty "
   457→                f"response body. This typically means the URL path is malformed or the "
   458→                f"model doesn't exist on this account. Verify the URL structure and model ID."
   459→            )
   460→        elif "model" in body_lower and ("not found" in body_lower or "invalid" in body_lower):
   461→            logger.error(f"  DIAGNOSIS: INVALID_MODEL — model ID not available on this account/region")
   462→        elif "payload" in body_lower or "body" in body_lower:
   463→            logger.error(f"  DIAGNOSIS: MALFORMED_REQUEST — request payload is invalid")
   464→        elif "header" in body_lower:
   465→            logger.error(f"  DIAGNOSIS: HEADER_FORMAT_ERROR — required header missing or malformed")
   466→        else:
   467→            logger.error(f"  DIAGNOSIS: BAD_REQUEST — check model ID and request format")
   468→
   469→    # Check for common key format issues
   470→    auth_header = headers_sent.get("Authorization", "")
   471→    if auth_header.startswith("Bearer "):
   472→        token = auth_header[7:]
   473→        if token != token.strip():
   474→            logger.error(f"  KEY_ISSUE: Bearer token has leading/trailing whitespace!")
   475→        if "\n" in token or "\r" in token:
   476→            logger.error(f"  KEY_ISSUE: Bearer token contains newline characters!")
   477→        if len(token) < 10:
   478→            logger.error(f"  KEY_ISSUE: Bearer token appears too short (len={len(token)})")
   479→

   513→class ProviderCircuitBreaker:
   514→    """Provider-level circuit breaker with automatic recovery.
   515→
   516→    Tracks overall provider health across all slots. When the failure count
   517→    exceeds the threshold, the circuit opens and rejects requests until the
   518→    recovery timeout elapses, at which point it transitions to half-open
   519→    and allows one request through to test recovery.
   520→    """
   521→
   522→    def __init__(
   523→        self,
   524→        provider_name: str,
   525→        failure_threshold: int = 20,
   526→        recovery_timeout: float = 300.0,
   527→    ):
   528→        self.provider_name = provider_name
   529→        self.failure_threshold = failure_threshold
   530→        self.recovery_timeout = recovery_timeout
   531→        self.failure_count = 0
   532→        self.last_failure_time = 0.0
   533→        self.state = "closed"  # closed, open, half_open
   534→
   535→    def record_success(self):
   536→        """Record a successful request — resets failure count and closes circuit."""
   537→        self.failure_count = 0
   538→        self.state = "closed"
   539→
   540→    def record_failure(self):
   541→        """Record a failed request — increments count and opens circuit if threshold reached."""
   542→        self.failure_count += 1
   543→        self.last_failure_time = time.time()
   544→        if self.failure_count >= self.failure_threshold:
   545→            if self.state != "open":
   546→                logger.warning(
   547→                    f"[{self.provider_name}] Circuit breaker OPENED — "
   548→                    f"{self.failure_count} consecutive failures reached threshold "
   549→                    f"({self.failure_threshold}). Will retry after "
   550→                    f"{self.recovery_timeout}s recovery timeout."
   551→                )
   552→            self.state = "open"
   553→
   554→    def allow_request(self) -> bool:
   555→        """Check if a request is allowed based on current circuit state."""
   556→        if self.state == "closed":
   557→            return True
   558→        if self.state == "open":
   559→            if time.time() - self.last_failure_time > self.recovery_timeout:
   560→                self.state = "half_open"
   561→                logger.info(
   562→                    f"[{self.provider_name}] Circuit breaker → HALF_OPEN — "
   563→                    f"recovery timeout elapsed, allowing test request"
   564→                )
   565→                return True
   566→            return False
   567→        return True  # half_open allows one request through

   585→    @staticmethod
   586→    def _post_json_with_retry(
   587→        url: str,
   588→        headers: dict,
   589→        payload: dict,
   590→        timeout: int,
   591→        provider_name: str = "unknown",
   592→        slot_index: int = 0,
   593→        max_retries: int = MAX_NETWORK_RETRIES,
   594→    ) -> Tuple[dict, float]:
   595→        """
   596→        Send a POST request with exponential backoff retry on network/retryable errors.
   597→        Logs verbose diagnostics on 403/400 auth failures (with key masking).
   598→        """
   599→        last_error = None
   600→
   601→        # Ensure User-Agent is set (Cloudflare blocks requests without it)
   602→        if "User-Agent" not in headers:
   603→            headers["User-Agent"] = _USER_AGENT
   604→
   605→        for attempt in range(max_retries + 1):
   606→            t0 = time.monotonic()
   607→            data = json.dumps(payload).encode("utf-8")
   608→            req = urllib.request.Request(url, data=data, headers=headers, method="POST")
   609→
   610→            try:
   611→                with urllib.request.urlopen(req, timeout=timeout) as resp:
   612→                    result = json.loads(resp.read().decode("utf-8"))
   613→                latency_ms = (time.monotonic() - t0) * 1000.0
   614→                return result, latency_ms
   615→
   616→            except urllib.error.HTTPError as e:
   617→                latency_ms = (time.monotonic() - t0) * 1000.0
   618→                error_body = _read_error_body(e)
   619→
   620→                # Cloudflare bot protection (error code 1010) — RETRYABLE
   621→                # This is NOT a real auth failure; it's transient bot detection
   622→                if e.code == 403 and _CF_BOT_ERROR_CODE in error_body:
   623→                    _log_auth_failure(provider_name, slot_index, e, url, headers)
   624→                    if attempt < max_retries:
   625→                        delay = _compute_backoff_delay(attempt) * 2  # extra delay for bot protection
   626→                        logger.warning(
   627→                            f"[{provider_name}] slot {slot_index} Cloudflare bot protection "
   628→                            f"(1010) — retry {attempt + 1}/{max_retries} in {delay:.1f}s"
   629→                        )
   630→                        time.sleep(delay)
   631→                        continue
   632→                    else:
   633→                        logger.error(
   634→                            f"[{provider_name}] slot {slot_index} Cloudflare bot protection "
   635→                            f"persisted after {max_retries + 1} attempts"
   636→                        )
   637→                        raise
   638→
   639→                # Auth failures (401, 403) — NEVER retry, log verbose diagnostics
   640→                # 401 Unauthorized = invalid/expired credentials (permanent)
   641→                # 403 Forbidden = revoked/insufficient permissions (permanent)
   642→                # Both are authentication/authorization failures that retrying
   643→                # will NEVER fix — they require credential rotation.
   644→                if e.code in AUTH_FAILURE_HTTP_CODES:
   645→                    _log_auth_failure(provider_name, slot_index, e, url, headers)
   646→                    logger.error(
   647→                        f"[{provider_name}] slot {slot_index} HTTP {e.code} — "
   648→                        f"AUTH FAILURE, NOT retrying (credential issue, not transient)"
   649→                    )
   650→                    raise
   651→
   652→                # 400 Bad Request — typically invalid model or malformed request
   653→                # Also NOT retried (the request itself is wrong, retrying won't help)
   654→                # BUT this is NOT an auth failure — raise BadRequestError instead.
   655→                if e.code == 400:
   656→                    logger.error(
   657→                        f"[{provider_name}] slot {slot_index} HTTP 400 — "
   658→                        f"BAD_REQUEST, NOT retrying (invalid model or malformed payload). "
   659→                        f"Check model ID and URL path."
   660→                    )
   661→                    raise BadRequestError(
   662→                        f"HTTP 400 for slot {slot_index}: "
   663→                        f"{error_body[:200] if error_body else 'empty body'}",
   664→                        provider=provider_name,
   665→                        slot=slot_index,
   666→                    )
   667→
   668→                # Retryable errors (429, 5xx)
   669→                if e.code in RETRYABLE_HTTP_CODES:
   670→                    if attempt < max_retries:
   671→                        delay = _compute_backoff_delay(attempt)
   672→                        logger.warning(
   673→                            f"[{provider_name}] slot {slot_index} HTTP {e.code} — "
   674→                            f"retry {attempt + 1}/{max_retries} in {delay:.1f}s"
   675→                        )
   676→                        time.sleep(delay)
   677→                        continue
   678→                    else:
   679→                        logger.error(
   680→                            f"[{provider_name}] slot {slot_index} HTTP {e.code} — "
   681→                            f"all {max_retries + 1} attempts exhausted"
   682→                        )
   683→                        raise
   684→
   685→                # Non-retryable errors (404, 405, etc.)
   686→                raise
   687→
   688→            except (urllib.error.URLError, ConnectionError, TimeoutError, OSError) as e:
   689→                latency_ms = (time.monotonic() - t0) * 1000.0
   690→                if attempt < max_retries:
   691→                    delay = _compute_backoff_delay(attempt)
   692→                    logger.warning(
   693→                        f"[{provider_name}] slot {slot_index} network error: {e} — "
   694→                        f"retry {attempt + 1}/{max_retries} in {delay:.1f}s"
   695→                    )
   696→                    time.sleep(delay)
   697→                    continue
   698→                else:
   699→                    logger.error(
   700→                        f"[{provider_name}] slot {slot_index} network error — "
   701→                        f"all {max_retries + 1} attempts exhausted: {e}"
   702→                    )
   703→                    raise
   704→
   705→        # Should not reach here, but just in case
   706→        raise last_error if last_error else RuntimeError("Unexpected retry loop exit")
   707→
   708→    @staticmethod
   709→    def _extract_text(response: dict) -> str:
   710→        """Extract text content from any provider response format.
   711→
   712→        Handles OpenAI-compatible format, CF Workers AI format, and legacy CF
   713→        format. NEVER returns str(response) — always returns a proper string
   714→        extracted from the response content, or empty string if extraction fails.
   715→        """
   716→        if not isinstance(response, dict):
   717→            return ""
   718→        # Format 1: OpenAI-compatible (choices[0].message.content)
   719→        try:
   720→            choices = response.get("choices", [])
   721→            if choices:
   722→                content = choices[0].get("message", {}).get("content", "")
   723→                if isinstance(content, str) and content.strip():
   724→                    return content.strip()
   725→        except (KeyError, IndexError, TypeError, AttributeError):
   726→            pass
   727→        # Format 2: CF Workers AI wrapped in 'result'
   728→        result = response.get("result", None)
   729→        if isinstance(result, dict):
   730→            # result.choices — nested OpenAI format
   731→            try:
   732→                r_choices = result.get("choices", [])
   733→                if r_choices:
   734→                    content = r_choices[0].get("message", {}).get("content", "")
   735→                    if isinstance(content, str) and content.strip():
   736→                        return content.strip()
   737→            except (KeyError, IndexError, TypeError, AttributeError):
   738→                pass
   739→            # result.response — legacy CF format
   740→            try:
   741→                resp_val = result.get("response", "")
   742→                if isinstance(resp_val, str) and resp_val.strip():
   743→                    return resp_val.strip()
   744→            except (KeyError, TypeError, AttributeError):
   745→                pass
   746→        elif isinstance(result, str) and result.strip():
   747→            return result.strip()
   748→        # Format 3: Empty or unrecognized — return empty string, NOT str(response)
   749→        logger.debug(
   750→            f"[_extract_text] Could not extract text from response keys: "
   751→            f"{list(response.keys()) if isinstance(response, dict) else 'non-dict'}"
   752→        )
   753→        return ""

   780→        # so the health check marks this provider as 'skipped'.

   783→        self._invalid_key_slots: list[int] = []

   790→            # Portkey keys are alphanumeric with no mandatory prefix.
   791→            effective_key = key or virtual_key
   792→            if len(effective_key) >= MIN_KEY_LEN:
   793→                self._active_slots.append(i)
   794→            else:
   795→                self._invalid_key_slots.append(i)

   797→                    f"[Portkey] slot {i} skipped — key too short "
   798→                    f"(len={len(effective_key)}, expected >={MIN_KEY_LEN})"
   799→                )

   802→            reason = (

   804→                f"(len < {MIN_KEY_LEN}). "
   805→                "Check PORTKEY_API_KEY_1/2/3 in GitHub Secrets."
   806→            )
   807→            logger.warning(f"[Portkey] {reason}")
   808→            raise ProviderConfigurationError(reason, provider="portkey")
   809→
   810→        logger.info(
   811→            f"[Portkey] Initialized with gateway: {_mask_url(self.gateway_url)} "

   813→            f"{len(self._invalid_key_slots)} invalid-format slot(s))"
   814→        )

   816→    @staticmethod
   817→    def _build_portkey_auth(slot: int) -> dict:
   818→        """
   819→        Build Portkey authentication headers with intelligent key detection.
   820→
   821→        Supports auth methods:
   822→        1. Native Portkey key (pk- prefix) → x-portkey-api-key header
   823→        2. Virtual key fallback (pk- prefix in PORTKEY_VIRTUAL_KEY) → combined auth
   824→        3. Provider API key (sk- prefix like OpenAI) → Bearer + x-portkey-provider
   825→        4. Generic alphanumeric key (no prefix) → Bearer auth with x-portkey-api-key
   826→
   827→        Raises ValueError if no valid key format is found.
   828→        """
   829→        key = os.environ.get(f"PORTKEY_API_KEY_{slot}", "").strip()
   830→        virtual_key = os.environ.get(f"PORTKEY_VIRTUAL_KEY_{slot}", "").strip()
   831→
   832→        headers = {"Content-Type": "application/json"}
   833→
   834→        if key.startswith("pk-"):
   835→            # Native Portkey key
   836→            headers["x-portkey-api-key"] = key
   837→            logger.debug(
   838→                f"[Portkey] slot {slot} Using native Portkey key: {_mask_key(key)}"
   839→            )
   840→        elif virtual_key.startswith("pk-"):
   841→            # Virtual key fallback
   842→            headers["x-portkey-api-key"] = virtual_key
   843→            headers["x-portkey-virtual-key"] = virtual_key
   844→            logger.debug(
   845→                f"[Portkey] slot {slot} Using virtual key: {_mask_key(virtual_key)}"
   846→            )
   847→        elif key.startswith("sk-"):
   848→            # Provider API key (e.g. OpenAI key) — route directly
   849→            headers["Authorization"] = f"Bearer {key}"
   850→            headers["x-portkey-provider"] = "openai"
   851→            logger.debug(
   852→                f"[Portkey] slot {slot} Using provider key (sk- prefix) "
   853→                f"with x-portkey-provider=openai"
   854→            )
   855→        elif key:
   856→            # Generic alphanumeric key (e.g. g8V...qTF) — try as Bearer with
   857→            # x-portkey-api-key header. Real Portkey keys may not have a prefix.
   858→            headers["Authorization"] = f"Bearer {key}"
   859→            headers["x-portkey-api-key"] = key
   860→            logger.debug(
   861→                f"[Portkey] slot {slot} Using generic key "
   862→                f"(starts with '{key[:4]}...') — attempting Bearer + x-portkey-api-key auth."
   863→            )
   864→        else:
   865→            raise ValueError(
   866→                f"Slot {slot}: no valid Portkey key found — "
   867→                f"PORTKEY_API_KEY_{slot} and PORTKEY_VIRTUAL_KEY_{slot} are empty"
   868→            )
   869→
   870→        # Also check for x-portkey-config header (virtual key config ID)
   871→        config_id = os.environ.get(
   872→            f"PORTKEY_CONFIG_{slot}",
   873→            os.environ.get("PORTKEY_CONFIG", "")
   874→        ).strip()
   875→        if config_id:
   876→            headers["x-portkey-config"] = config_id
   877→            logger.debug(
   878→                f"[Portkey] slot {slot} Using config: {_mask_key(config_id)}"
   879→            )
   880→
   881→        return headers

   885→        """Validate Portkey API key format and return list of diagnostic issues.

   887→        Portkey keys may have various formats: pk-xxx-xxx (native),

   902→        if key != key.strip():
   903→            issues.append("Key has leading/trailing whitespace")
   904→        if issues:
   905→            for issue in issues:
   906→                logger.warning(
   907→                    f"[Portkey] slot {slot_index} KEY VALIDATION: {issue}"
   908→                )
   909→        return issues

   923→    def chat_complete(
   924→        self, messages, model=None, max_tokens=2048, temperature=0.2,
   925→        timeout=60, task="general"
   926→    ) -> str:
   927→        # Provider-level circuit breaker check
   928→        if not self.circuit_breaker.allow_request():
   929→            logger.warning(
   930→                f"[Portkey] Circuit breaker OPEN — skipping request"
   931→            )
   932→            raise RuntimeError(
   933→                f"Portkey provider circuit breaker is OPEN "
   934→                f"({self.circuit_breaker.failure_count} consecutive failures)"
   935→            )
   936→
   937→        explicit_model = model
   938→        chosen_model   = explicit_model or self.DEFAULT_MODEL
   939→        models_to_try  = [chosen_model] + [m for m in self.PORTKEY_MODELS if m != chosen_model]
   940→
   941→        slot      = self.rotator.get_primary()
   942→        fallbacks = [slot] + self.rotator.get_fallback_chain(slot.index)
   943→
   944→        last_auth_error = None
   945→        for attempt, s in enumerate(fallbacks):
   946→            last_err = None
   947→            for m in models_to_try:
   948→                try:
   949→                    # Sanitize API key
   950→                    clean_key = _sanitize_api_key(s.api_key)
   951→                    if not clean_key:
   952→                        logger.warning(f"[Portkey] slot {s.index} has empty API key — skipping")
   953→                        break  # No point trying other models with empty key
   954→
   955→                    # Validate key format
   956→                    key_issues = self._validate_portkey_key(clean_key, s.index)
   957→                    if key_issues:
   958→                        logger.warning(
   959→                            f"[Portkey] slot {s.index} Key format issues detected. "
   960→                            f"If auth fails, check: (1) key starts with 'pk-', "
   961→                            f"(2) key is not expired, (3) PORTKEY_VIRTUAL_KEY_{s.index} "
   962→                            f"env var as alternative auth."
   963→                        )
   964→
   965→                    url     = f"{self.gateway_url}/chat/completions"
   966→
   967→                    # ── Portkey Authentication Strategy ─────────────────────
   968→                    # Use the enhanced _build_portkey_auth method for intelligent
   969→                    # key detection: pk- (native), virtual key, sk- (provider),
   970→                    # or unknown format with warning.
   971→                    try:
   972→                        headers = self._build_portkey_auth(s.index)
   973→                    except ValueError as auth_err:
   974→                        logger.warning(
   975→                            f"[Portkey] slot {s.index} Auth build failed: {auth_err}"
   976→                        )
   977→                        break  # No point trying other models with bad auth
   978→
   979→                    payload = {
   980→                        "model":       m,
   981→                        "messages":    messages,
   982→                        "max_tokens":  max_tokens,
   983→                        "temperature": temperature,
   984→                    }
   985→                    resp, lat = self._post_json_with_retry(
   986→                        url, headers, payload, timeout,
   987→                        provider_name="Portkey", slot_index=s.index
   988→                    )
   989→                    self.rotator.mark_success(s, lat)
   990→                    self.circuit_breaker.record_success()
   991→                    return self._extract_text(resp)
   992→                except BadRequestError as e:
   993→                    # HTTP 400 — NOT an auth failure, try next model
   994→                    logger.debug(
   995→                        f"[Portkey] slot {s.index} model {m} → "
   996→                        f"BadRequestError: {str(e)[:200]}"
   997→                    )
   998→                    continue  # Try next model
   999→                except urllib.error.HTTPError as e:
  1000→                    last_err = e
  1001→                    if e.code in (403, 401):
  1002→                        last_auth_error = e

  1748→        if model:

  1752→            logger.debug(f"[CF-AI-GW] Dynamic model [{task}]: {selected}")
  1753→            return selected

  1758→    def chat_complete(
  1759→        self, messages, model=None, max_tokens=2048, temperature=0.2,
  1760→        timeout=60, task="general"
  1761→    ) -> str:
  1762→        # Provider-level circuit breaker check
  1763→        if not self.circuit_breaker.allow_request():
  1764→            logger.warning(
  1765→                f"[CF-AI-GW] Circuit breaker OPEN — skipping request"
  1766→            )
  1767→            raise RuntimeError(
  1768→                f"CF-AI-GW provider circuit breaker is OPEN "
  1769→                f"({self.circuit_breaker.failure_count} consecutive failures)"
  1770→            )
  1771→
  1772→        chosen_model = self._resolve_model(model, task)
  1773→
  1774→        slot      = self.rotator.get_primary()
  1775→        fallbacks = [slot] + self.rotator.get_fallback_chain(slot.index)
  1776→
  1777→        models_to_try = [chosen_model] + [m for m in CF_STABLE_MODELS if m != chosen_model]
  1778→
  1779→        last_auth_error = None
  1780→        for attempt, s in enumerate(fallbacks):
  1781→            last_err = None
  1782→
  1783→            # Skip dead slots (slots that previously returned 400+empty body)
  1784→            with self._dead_slots_lock:
  1785→                if s.index in self._dead_slots:
  1786→                    continue
  1787→
  1788→            # Probe gateway reachability before first attempt on this slot
  1789→            if attempt == 0 or last_auth_error is None:
  1790→                if not self._probe_gateway(s.gateway_url):
  1791→                    logger.warning(
  1792→                        f"[CF-AI-GW] slot {s.index} gateway unreachable — skipping"
  1793→                    )
  1794→                    self.rotator.mark_failure(s)
  1795→                    self.circuit_breaker.record_failure()
  1796→                    continue
  1797→
  1798→            for m in models_to_try:
  1799→                # Skip models that already 400'd on a previous slot
  1800→                if m in self._failed_models:
  1801→                    logger.debug(
  1802→                        f"[CF-AI-GW] Skipping model {m} — previously 400'd on another slot"
  1803→                    )
  1804→                    continue
  1805→
  1806→                try:
  1807→                    # Sanitize credentials
  1808→                    clean_token = _sanitize_api_key(s.api_key)
  1809→                    clean_acct  = _sanitize_api_key(s.account_id)
  1810→                    if not clean_token:
  1811→                        logger.warning(
  1812→                            f"[CF-AI-GW] slot {s.index} has empty API token — skipping"
  1813→                        )
  1814→                        break  # No point trying other models with empty token
  1815→
  1816→                    # CORRECTION 7: Use OpenAI-compatible endpoint
  1817→                    # Model ID goes in request body, NOT URL path.
  1818→                    # URL: {gateway_base}/workers-ai/v1/chat/completions
  1819→                    url = self._build_cf_gateway_url(s.gateway_url)
  1820→                    headers = {
  1821→                        "Content-Type":  "application/json",
  1822→                        "Authorization": f"Bearer {clean_token}",
  1823→                    }
  1824→                    payload = self._build_cf_request_body(
  1825→                        model_id=m,
  1826→                        messages=messages,
  1827→                        max_tokens=max_tokens,
  1828→                        temperature=temperature,
  1829→                    )
  1830→                    resp, lat = self._post_json_with_retry(
  1831→                        url, headers, payload, timeout,
  1832→                        provider_name="CF-AI-GW", slot_index=s.index
  1833→                    )
  1834→                    self.rotator.mark_success(s, lat)
  1835→                    self.circuit_breaker.record_success()
  1836→                    return CloudflareWorkersAIProvider._extract_cf_content(resp)
  1837→                except BadRequestError as e:
  1838→                    # HTTP 400 — NOT an auth failure. Check if empty body (dead slot)
  1839→                    # or model-specific issue (try next model).
  1840→                    err_msg = str(e)
  1841→                    if "empty body" in err_msg.lower():
  1842→                        with self._dead_slots_lock:
  1843→                            if s.index not in self._dead_slots:
  1844→                                self._dead_slots.add(s.index)
  1845→                                logger.warning(
  1846→                                    f"[CF-AI-GW] slot {s.index} permanently failed "
  1847→                                    f"(HTTP 400 empty body) — "
  1848→                                    f"skipping all remaining models for this slot"
  1849→                                )
  1850→                        break  # Stop trying models for this slot
  1851→                    # Model-specific issue — add to failed set and try next model
  1852→                    self._failed_models.add(m)
  1853→                    logger.debug(
  1854→                        f"[CF-AI-GW] slot {s.index} model {m} → "
  1855→                        f"BadRequestError: {err_msg[:200]}"
  1856→                    )
  1857→                    continue
  1858→                except urllib.error.HTTPError as e:
  1859→                    last_err = e
  1860→                    if e.code in (403, 401):
  1861→                        last_auth_error = e
  1862→                        logger.warning(
  1863→                            f"[CF-AI-GW] slot {s.index} AUTH FAIL HTTP {e.code} "
  1864→                            f"URL={s.gateway_url[:40]}..."
  1865→                        )
  1866→                        self.circuit_breaker.record_failure()
  1867→                        break  # Try next slot, not next model
  1868→                    # NOTE: 400 is now caught as BadRequestError above, not here
  1869→                    if e.code == 404:
  1870→                        logger.debug(f"[CF-AI-GW] Model not found: {m}")
  1871→                        self._failed_models.add(m)
  1872→                        continue
  1873→                    logger.warning(
  1874→                        f"[CF-AI-GW] slot {s.index} "
  1875→                        f"URL={s.gateway_url[:40]}... HTTP {e.code}"
  1876→                    )
  1877→                    self.circuit_breaker.record_failure()
  1878→                    raise
  1879→
  1880→            if last_err:
  1881→                if last_err.code in (403, 401):
  1882→                    self.rotator.mark_failure(s)
  1883→                    continue  # Try next slot
  1884→                logger.warning(f"[CF-AI-GW] slot {s.index} all models failed")
  1885→                self.rotator.mark_failure(s)
  1886→                self.circuit_breaker.record_failure()
  1887→                if attempt == len(fallbacks) - 1:
  1888→                    raise last_err
  1889→                time.sleep(2 ** attempt)
  1890→
  1891→        # All slots exhausted — check if all are dead (config error, not runtime)
  1892→        with self._dead_slots_lock:
  1893→            all_dead = all(
  1894→                s.index in self._dead_slots
  1895→                for s in self.rotator.slots
  1896→            )
  1897→
  1898→        if all_dead:
  1899→            raise ProviderConfigurationError(
  1900→                f"[CF-AI-GW] All {len(self.rotator.slots)} slots failed "
  1901→                f"(HTTP 400 / empty body on first model attempt per slot). "
  1902→                f"Verify CF_ACCOUNT_ID and CF_API_TOKEN values in GitHub Secrets.",
  1903→                provider="cloudflare_ai_gateway",
  1904→            )
  1905→
  1906→        if last_auth_error:
  1907→            raise last_auth_error
  1908→        return ""
