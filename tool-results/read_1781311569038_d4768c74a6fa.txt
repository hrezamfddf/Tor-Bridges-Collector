   156‚Üí   156‚Üídef _preflight_screen_slot(slot_index: int) -> Tuple[bool, str]:
   157‚Üí   157‚Üí    """
   158‚Üí   158‚Üí    Enhanced pre-flight screening for Cloudflare slots (Amendment 6).
   159‚Üí   159‚Üí
   160‚Üí   160‚Üí    Validates account_id format (32-char hex), API token length (>=40 chars),
   161‚Üí   161‚Üí    and gateway URL structure BEFORE sending any request.
   162‚Üí   162‚Üí
   163‚Üí   163‚Üí    Broken slots (like slot 7 with corrupted tokens) are detected here
   164‚Üí   164‚Üí    and silently skipped without causing HTTP 400 errors.
   165‚Üí   165‚Üí
   166‚Üí   166‚Üí    Returns:
   167‚Üí   167‚Üí        (valid: bool, reason: str) ‚Äî valid=True means slot is usable.
   168‚Üí   168‚Üí    """
   169‚Üí   169‚Üí    account_id  = os.environ.get(f"CF_ACCOUNT_ID_{slot_index}", "").strip()
   170‚Üí   170‚Üí    api_token   = os.environ.get(f"CF_API_TOKEN_{slot_index}", "").strip()
   171‚Üí   171‚Üí    gateway_url = os.environ.get(f"CF_AI_GATEWAY_URL_{slot_index}", "").strip()
   172‚Üí   172‚Üí
   173‚Üí   173‚Üí    # Rule 1: Both must be non-empty
   174‚Üí   174‚Üí    if not account_id or not api_token:
   175‚Üí   175‚Üí        return False, "missing credentials"
   176‚Üí   176‚Üí
   177‚Üí   177‚Üí    # Rule 2: Account ID must be 32-char hex
   178‚Üí   178‚Üí    if not re.match(r'^[0-9a-f]{32}$', account_id, re.IGNORECASE):
   179‚Üí   179‚Üí        return False, f"invalid account_id format (len={len(account_id)})"
   180‚Üí   180‚Üí
   181‚Üí   181‚Üí    # Rule 3: API token must be >=40 chars (CF tokens are 40+)
   182‚Üí   182‚Üí    if len(api_token) < CF_MIN_TOKEN_LENGTH:
   183‚Üí   183‚Üí        return False, f"token too short ({len(api_token)} chars, min={CF_MIN_TOKEN_LENGTH})"
   184‚Üí   184‚Üí
   185‚Üí   185‚Üí    # Rule 4: Gateway URL must match CF pattern (if provided)
   186‚Üí   186‚Üí    if gateway_url:
   187‚Üí   187‚Üí        pattern = r'^https://gateway\.ai\.cloudflare\.com/v1/[0-9a-f]{32}/'
   188‚Üí   188‚Üí        if not re.match(pattern, gateway_url, re.IGNORECASE):
   189‚Üí   189‚Üí            return False, "malformed gateway URL"
   190‚Üí   190‚Üí
   191‚Üí   191‚Üí    # Rule 5: Account ID in gateway URL must match credentials
   192‚Üí   192‚Üí    if gateway_url and account_id:
   193‚Üí   193‚Üí        # Extract account_id from gateway URL
   194‚Üí   194‚Üí        gw_match = re.search(r'/v1/([0-9a-f]{32})/', gateway_url, re.IGNORECASE)
   195‚Üí   195‚Üí        if gw_match:

   198‚Üí   198‚Üí                return False, "account_id mismatch between URL and credentials"

   462‚Üí   462‚Üí        elif "payload" in body_lower or "body" in body_lower:
   463‚Üí   463‚Üí            logger.error(f"  DIAGNOSIS: MALFORMED_REQUEST ‚Äî request payload is invalid")

   513‚Üí   513‚Üíclass ProviderCircuitBreaker:
   514‚Üí   514‚Üí    """Provider-level circuit breaker with automatic recovery.
   515‚Üí   515‚Üí
   516‚Üí   516‚Üí    Tracks overall provider health across all slots. When the failure count
   517‚Üí   517‚Üí    exceeds the threshold, the circuit opens and rejects requests until the
   518‚Üí   518‚Üí    recovery timeout elapses, at which point it transitions to half-open
   519‚Üí   519‚Üí    and allows one request through to test recovery.
   520‚Üí   520‚Üí    """
   521‚Üí   521‚Üí
   522‚Üí   522‚Üí    def __init__(
   523‚Üí   523‚Üí        self,
   524‚Üí   524‚Üí        provider_name: str,
   525‚Üí   525‚Üí        failure_threshold: int = 20,
   526‚Üí   526‚Üí        recovery_timeout: float = 300.0,
   527‚Üí   527‚Üí    ):
   528‚Üí   528‚Üí        self.provider_name = provider_name
   529‚Üí   529‚Üí        self.failure_threshold = failure_threshold
   530‚Üí   530‚Üí        self.recovery_timeout = recovery_timeout
   531‚Üí   531‚Üí        self.failure_count = 0
   532‚Üí   532‚Üí        self.last_failure_time = 0.0
   533‚Üí   533‚Üí        self.state = "closed"  # closed, open, half_open
   534‚Üí   534‚Üí
   535‚Üí   535‚Üí    def record_success(self):
   536‚Üí   536‚Üí        """Record a successful request ‚Äî resets failure count and closes circuit."""
   537‚Üí   537‚Üí        self.failure_count = 0
   538‚Üí   538‚Üí        self.state = "closed"
   539‚Üí   539‚Üí
   540‚Üí   540‚Üí    def record_failure(self):
   541‚Üí   541‚Üí        """Record a failed request ‚Äî increments count and opens circuit if threshold reached."""
   542‚Üí   542‚Üí        self.failure_count += 1
   543‚Üí   543‚Üí        self.last_failure_time = time.time()
   544‚Üí   544‚Üí        if self.failure_count >= self.failure_threshold:
   545‚Üí   545‚Üí            if self.state != "open":
   546‚Üí   546‚Üí                logger.warning(
   547‚Üí   547‚Üí                    f"[{self.provider_name}] Circuit breaker OPENED ‚Äî "
   548‚Üí   548‚Üí                    f"{self.failure_count} consecutive failures reached threshold "
   549‚Üí   549‚Üí                    f"({self.failure_threshold}). Will retry after "
   550‚Üí   550‚Üí                    f"{self.recovery_timeout}s recovery timeout."
   551‚Üí   551‚Üí                )
   552‚Üí   552‚Üí            self.state = "open"
   553‚Üí   553‚Üí
   554‚Üí   554‚Üí    def allow_request(self) -> bool:
   555‚Üí   555‚Üí        """Check if a request is allowed based on current circuit state."""
   556‚Üí   556‚Üí        if self.state == "closed":
   557‚Üí   557‚Üí            return True
   558‚Üí   558‚Üí        if self.state == "open":
